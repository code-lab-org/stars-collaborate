\[ \vec{p} = Position \]

\[ \hat{r} = Unit~Focus~Direction \]

\[ a = b = EARTH~semimajor~axis,~ c = EARTH~semiminor~axis \]

\[ \vec{q}_1=\begin{bmatrix} 1 \div a&1 \div b&1 \div c\end{bmatrix} \]

\[ \vec{q}_2=\begin{bmatrix} a& b& c\end{bmatrix} \]

\[ \vec{p}_{q} = \vec{p}\vec{q}_1 \]

\[ Quadratic~Formula \]

\[ A = \left\lvert \vec{r} \right\rvert^2 \]

\[ B = 2(\vec{p}_{q} \cdot \vec{r}) \]

\[ C = \left\lvert\vec{p}_{q}\right\rvert^2 - 1 \]

\[ t_1 = \frac{-B+\sqrt{B^2 - 4AC}}{2A} \]

\[ t_2 = \frac{-B-\sqrt{B^2 - 4AC}}{2A} \]

\[ Intersection = \min{\left(\left(\vec{p}_{q}-((\vec{p}_{q}+\hat{r}*t_1)*\vec{q}_2)\right), \left(\vec{p}_{q} - ((\vec{p}_{q}+\hat{r}*t_2)*\vec{q}_2)\right)\right)} \]
